#테이블 생성 전에 yam데이터베이스 생성
create database yam;

#user 테이블
create table user(
   `user_no` int NOT NULL auto_increment,
   `user_id` varchar(20) NOT NULL,
   `password` varchar(20) NOT NULL,
   `user_name` varchar(12) NOT NULL,
   `email` varchar(40) NOT NULL,
   primary key(`user_no`)
);

#회원가입 시 user_no 1부터 다시 증가 (행 삭제 시 이용)
ALTER TABLE user AUTO_INCREMENT=1;

#roomlist 테이블
create table roomlist(
   `room_no` int NOT NULL,
   `room_name` varchar(40) NOT NULL,
   `password` varchar(10),
   `game_name` varchar(40) NOT NULL,
   `nowplayer` int NOT NULL,
   `maxplayer` int NOT NULL,
   `createtime` datetime,
   primary key(`room_no`)
);

#프로시저 생성
DELIMITER $$
DROP PROCEDURE IF EXISTS roomlistinit$$
CREATE PROCEDURE roomlistinit(a int)
BEGIN
   DELETE FROM roomlist;
	WHILE a < 100 DO
		INSERT INTO roomlist VALUES (a, 'default', NULL, 'default', 0, 0, now());
		SET a = a + 1;
	END WHILE;
END$$
DELIMITER ;

#프로시저 실행
CALL roomlistinit(1);

#roomuser 테이블
create table roomuser(
   `user_no` int NOT NULL,
   `user_name` varchar(12) NOT NULL,
   `room_no` int NOT NULL,
   `ready` int,
   `turn` int,
   primary key(`user_no`)
);

#chat 테이블
create table chatting(
   `room_no` int NOT NULL,
   `chat` varchar(50) NOT NULL,
   primary key(`room_no`)
);
